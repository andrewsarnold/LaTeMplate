% Page margins
\usepackage[top=1.4in, bottom=0.9in, left=0.9in, right=0.9in, headsep=3em]{geometry}

% Cover image
\usepackage{incgraph}

% Font
\usepackage[T1]{fontenc}
\renewcommand{\rmdefault}{ppl} % Serif type: Palatino
\renewcommand{\sfdefault}{pag} % Sans serif type: Avant Garde
\usepackage{sourcecodepro} % Monospace type: Source Code Pro

% Colors
\usepackage{color}
\definecolor{accent}{rgb}{0.078,0.506,0.721}

% Special font for section headers
\usepackage{sectsty}
\partfont{\sffamily\color{accent}}
\chapterfont{\sffamily\color{accent}}
\sectionfont{\sffamily\color{accent}}
\subsectionfont{\sffamily\color{accent}}
\subsubsectionfont{\sffamily\color{accent}}

% Other text formatting
\sloppy
\frenchspacing

% Don't number subsections
\setcounter{secnumdepth}{0}

% Headers / footers
\usepackage{fancyhdr}
\fancyhf{} % Clear the header and footer (we'll define our own below)
\renewcommand{\headrulewidth}{0pt} % Remove line at top of page

% Hyperref
\usepackage{hyperref}
\hypersetup{
	colorlinks,
	linkcolor={accent},
	citecolor={accent},
	urlcolor={accent}
}

% Parts
\newcommand{\apart}[1]{\part{#1}\label{#1}}

% Chapters that are regular chapters
\newcommand{\chap}[1]{
	\chapter{#1}
	\thispagestyle{empty}
	\chapfooter{#1}
}

% Chapters that aren't really chapters
\newcommand{\nonchap}[1]{
	\chapter*{#1}
	\addcontentsline{toc}{chapter}{#1}
	\thispagestyle{empty}
	\chapfooter{#1}
	\markboth{#1}{}
}

\newcommand{\chapfooter}[1]{\setcounter{footnote}{0}
\lhead[\fancyplain{}{\color{accent}\textbf{\thepage} \hspace{1em} \small{#1 \hspace{1em} \textbf{\nouppercase{\rightmark}}}}]{}
\rhead[]{\fancyplain{}{\color{accent}\small{\textbf{\nouppercase{\rightmark}} \hspace{1em} #1} \hspace{1em} \textbf{\thepage}}}}

% Symbolic footnotes
\usepackage{perpage}
\renewcommand{\thefootnote}{\fnsymbol{footnote}}
\MakePerPage{footnote}
